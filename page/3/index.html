# HashTable
class LinkList:
    class Node:
        def __init__(self, item=None):
            self.item = item
            self.next = None

    class LinkListIterator:
        def __init__(self, head):
            self.current = head

        def __iter__(self):
            return self

        def __next__(self):
            if self.current is None:
                raise StopIteration
            else:
                item = self.current.item
                self.current = self.current.next
                return item

    def __init__(self, iterable=None):
        self.head = None
        self.tail = None
        if iterable:
            self.extend(iterable)

    def append(self, obj):
        node = self.Node(obj)
        if self.head is None:
            self.head = node
            self.tail = node
        else:
            self.tail.next = node
            self.tail = node

    def extend(self, iterable):
        for item in iterable:
            self.append(item)

    def find(self, obj):
        for item in self:
            if item == obj:
                return True
        return False

    def __iter__(self):
        return self.LinkListIterator(self.head)

    def __repr__(self):
        return str(list(self))

class HashTable:
    def __init__(self, size=101):
        self.size = size
        self.table = [LinkList() for i in range(self.size)]

    def hash(self, obj):
        return obj % self.size

    def insert(self, obj):
        index = self.hash(obj)
        if self.find(obj):
            return False
        else:
            self.table[index].append(obj)
            return True

    def find(self, obj):
        index = self.hash(obj)
        return self.table[index].find(obj)

ht = HashTable()
ht.insert(0)
ht.insert(1)
ht.insert(101)
print(','.join(map(str, ht.table)))
print(ht.find(1))
print(ht.find(2))

# 二分查找
def binary_search(lst, val):
    left = 0
    right = len(lst) - 1
    while left < right:
        mid = (left + right) // 2
        if lst[mid] > val:
            right = mid - 1
        elif lst[mid] < val:
            left = mid + 1
        else:
            return mid
    return -1

li = list(range(100))
print(binary_search(li, 50))  # 50
print(binary_search(li, 200))  # -1

def count_sort(lst, max_count=100):
    count = [0 for _ in range(max_count + 1)]
    for val in lst:
        count[val] += 1
    lst.clear()
    for ind, val in enumerate(count):
        for i in range(val):
            lst.append(ind)

li = [4, 6, 2, 3, 1, 5, 7, 9, 8]
count_sort(li, 9)
print(li)  # [1, 2, 3, 4, 5, 6, 7, 8, 9]

时间复杂度：O(n)
空间复杂度：O(n)
稳定性：稳定

基数排序

def radix_sort(lst):
    max_num = max(lst)
    it = 0
    while 10 ** it <= max_num:
        buckets = [[] for _ in range(10)]
        for i in lst:
            buckets[i // 10 ** it % 10].append(i)
        lst.clear()
        for bucket in buckets:
            lst.extend(bucket)
        it += 1

li = [4, 6, 2, 3, 1, 5, 7, 9, 8]
radix_sort(li)
print(li)

时间复杂度：O(nm)
空间复杂度：O(n)
稳定性：稳定

希尔排序

def insert_sort_gap(lst, gap):
    for i in range(gap, len(lst)):
        tmp = lst[i]
        j = i - gap
        while j >= 0 and lst[j] > tmp:
            lst[gap + j] = lst[j]
            j -= gap
        lst[j + gap] = tmp

def shell_sort(lst):
    n = len(lst) // 2
    while n >= 1:
        insert_sort_gap(lst, n)
        n //= 2

li = [4, 6, 2, 3, 1, 5, 7, 9, 8]
shell_sort(li)
print(li)  # [1, 2, 3, 4, 5, 6, 7, 8, 9]

桶排序

def bucket_sort(lst, n=100, max_num=10000):
    buckets = [[] for _ in range(n)]
    for val in lst:
        i = min(val // (max_num // n), n - 1)
        buckets[i].append(val)
        for j in range(len(buckets[i]) - 1, 0, -1):
            if buckets[i][j - 1] > buckets[i][j]:
                buckets[i][j - 1], buckets[i][j] = buckets[i][j], buckets[i][j - 1]
            else:
                break
    sorted_lst = []
    for bucket in buckets:
        sorted_lst.extend(bucket)
    return sorted_lst

li = [4, 6, 2, 3, 1, 5, 7, 9, 8]
print(bucket_sort(li))  # [1, 2, 3, 4, 5, 6, 7, 8, 9]

时间复杂度：O(nlogm)
空间复杂度：O(logm)
稳定性：稳定

快速排序

def partition(lst, left, right):
    tmp = lst[left]
    while left < right:
        while left < right and lst[right] > tmp:
            right -= 1
        lst[left] = lst[right]
        while left < right and lst[left] < tmp:
            left += 1
        lst[right] = lst[left]
    lst[left] = tmp
    return left

def quick_sort(lst, left, right):
    if left < right:
        mid = partition(lst, left, right)
        quick_sort(lst, left, mid)
        quick_sort(lst, mid + 1, right)

li = [4, 6, 2, 3, 1, 5, 7, 9, 8]
quick_sort(li, 0, len(li) - 1)
print(li)  # [1, 2, 3, 4, 5, 6, 7, 8, 9]

partition:
对传入的lst部分进行分割，范围内的第一个数作为中间值，把小于中间值的置于左侧，大于的置于右侧，并返回中间值的下标
处理方法为第一次从right开始向左遍历，若遇到比中间值小的则将那个值置于left，然后从left向右遍历，遇到比中间值大的则置于right，以此类推
(优化)为防止每次传入的left为范围最大值，随机从范围内取一个数和第一个数替换
lst[(left+right)//2], lst[left] = lst[left], lst[(left+right)//2]

quick_sort:
将传入的列表递归通过二分的方法切片传入partition

时间复杂度：O(nlogn)
空间复杂度：O(logn)
稳定性：不稳定

堆排序

def sift(lst, low, high):
    i = low
    j = 2 * i + 1
    tmp = lst[i]
    while j <= high:
        if j + 1 <= high and lst[j + 1] > lst[j]:
            j += 1
        if lst[j] > tmp:
            lst[i] = lst[j]
            i = j
            j = 2 * i + 1
        else:
            lst[i] = tmp
            break
    else:
        lst[i] = tmp

def heap_sort(lst):
    n = len(lst)
    for i in range((n - 2) // 2, -1, -1):
        sift(lst, i, n - 1)
    for i in range(n - 1, -1, -1):
        lst[0], lst[i] = lst[i], lst[0]
        sift(lst, 0, i - 1)

li = [4, 6, 2, 3, 1, 5, 7, 9, 8]
heap_sort(li)
print(li)  # [1, 2, 3, 4, 5, 6, 7, 8, 9]

sift:
low为根节点，high为最后一个叶节点，tmp为根节点的值
定义i为low，j为i的左子节点，判断i的右子节点如果比j大则将j替换为i的右子节点
判断i和j的值，若j比i大则将i的值替换为j的值，并将j定义为新的i的左子节点并重复上一步骤
若i比j大，则将i的值替换为tmp并跳出循环；循环正常退出(j比high大)则将i的值替换为tmp

heap_sort:
1.构建堆，倒序遍历(i)将最后一个非根节点((n-2)//2)做向下调整(i——len(lst)-1)
2.倒序遍历无序区将最后一个叶节点与根节点交换，将无序区做一次向下调整，并将无序区最后一个值下标向前一位

时间复杂度：O(nlogn)
空间复杂度：O(1)
稳定性：不稳定

归并排序

def marge(lst, low, mid, high):
    i = low
    j = mid + 1
    tmp_lst = []
    while i <= mid and j <= high:
        if lst[i] < lst[j]:
            tmp_lst.append(lst[i])
            i += 1
        else:
            tmp_lst.append(lst[j])
            j += 1
    while i <= mid:
        tmp_lst.append(lst[i])
        i += 1
    while j <= high:
        tmp_lst.append(lst[j])
        j += 1
    lst[low:high + 1] = tmp_lst

def marge_sort(lst, low, high):
    if low < high:
        mid = (low + high) // 2
        marge_sort(lst, low, mid)
        marge_sort(lst, mid + 1, high)
        marge(lst, low, mid, high)

li = [4, 6, 2, 3, 1, 5, 7, 9, 8]
marge_sort(li, 0, len(li) - 1)
print(li)  # [1, 2, 3, 4, 5, 6, 7, 8, 9]

marge:
每次归并会传入low,mid,high，创建临时列表tmp_lst，定义i为low，j为mid+1
若i比j的值大，则将i下标的值添加到tmp_lst，并自增
反之则将j下标的值添加到tmp_lst，并自增
最后将lst需要归并的范围(low——high)的内容替换为tmp_lst

marge_sort:
将传入的lst递归分割为二等分，直到分割的长度为1为止，再将每两份分割的内容归并

时间复杂度：O(nlogn)
空间复杂度：O(n)
稳定性：稳定

冒泡排序

def bubble_sort(lst):
    for i in range(len(lst) - 1):
        flag = False
        for j in range(len(lst) - i - 1):
            if lst[j] > lst[j + 1]:
                lst[j], lst[j + 1] = lst[j + 1], lst[j]
                flag = True
        if not flag:
            return

li = [4, 6, 2, 3, 1, 5, 7, 9, 8]
bubble_sort(li)
print(li)  # [1, 2, 3, 4, 5, 6, 7, 8, 9]

若无序区的相邻两个数前面的数大于后面的则交换
(优化)每一趟结束后判断flag，若无修改则直接退出

i为第几趟(0开始)
j为遍历无序区的下标

有序区为0——(i-1)
无序区为i——(len(lst)-1)

时间复杂度：O(n²)
空间复杂度：O(1)
稳定性：稳定

插入排序

def insert_sort(lst):
    for i in range(1, len(lst)):
        tmp = lst[i]
        j = i-1
        while j >= 0 and lst[j] > tmp:
            lst[j+1] = lst[j]
            j -= 1
        lst[j+1] = tmp

li = [4, 6, 2, 3, 1, 5, 7, 9, 8]
insert_sort(li)
print(li)  # [1, 2, 3, 4, 5, 6, 7, 8, 9]

初始手里一张牌(lst[0])
每次摸到一张牌(lst[i])，判断i前面的牌(j)是否比摸到的牌大，大则将前面的牌向后移一位，否则j向前移动一位并进行下次判断
若lst[j]<lst[i]则将摸到的牌的值插入到lst[j+1]

i为无序区第一个数
j为有序区为以-1步进遍历的下标

有序区为0——(i-1)
无序区为i——(len(lst)-1)

时间复杂度：O(n²)
空间复杂度：O(1)
稳定性：稳定

选择排序

def select_sort(lst):
    for i in range(len(lst)-1):
        tmp = i
        for j in range(i, len(lst)):
            if lst[j] < lst[tmp]:
                tmp = j
        lst[i], lst[tmp] = lst[tmp], lst[i]

li = [4, 6, 2, 3, 1, 5, 7, 9, 8]
select_sort(li)
print(li)  # [1, 2, 3, 4, 5, 6, 7, 8, 9]

每一趟找出无序区的最小数并与无序区第一个数交换

i为第几趟(0开始)
j为遍历无序区的下标

有序区为0——(i-1)
无序区为i——(len(lst)-1)

时间复杂度：O(n²)
空间复杂度：O(1)
稳定性：不稳定
