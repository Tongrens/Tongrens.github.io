# HashTable
class LinkList:
    class Node:
        def __init__(self, item=None):
            self.item = item
            self.next = None

    class LinkListIterator:
        def __init__(self, head):
            self.current = head

        def __iter__(self):
            return self

        def __next__(self):
            if self.current is None:
                raise StopIteration
            else:
                item = self.current.item
                self.current = self.current.next
                return item

# 二分查找
def binary_search(lst, val):
    left = 0
    right = len(lst) - 1
    while left < right:
        mid = (left + right) // 2
        if lst[mid] > val:
            right = mid - 1
        elif lst[mid] < val:
            left = mid + 1
        else:
            return mid
    return -1

li = list(range(100))
print(binary_search(li, 50))  # 50
print(binary_search(li, 200))  # -1

def count_sort(lst, max_count=100):
    count = [0 for _ in range(max_count + 1)]
    for val in lst:
        count[val] += 1
    lst.clear()
    for ind, val in enumerate(count):
        for i in range(val):
            lst.append(ind)

li = [4, 6, 2, 3, 1, 5, 7, 9, 8]
count_sort(li, 9)
print(li)  # [1, 2, 3, 4, 5, 6, 7, 8, 9]

时间复杂度：O(n)
空间复杂度：O(n)
稳定性：稳定
