使用Powershell导入csv用户并批量创建

$users=Import-Csv 1.csv
foreach ($user in $users) {New-ADUser $user.Username -Surname $user."last name" -GivenName $user.'first Name' -EmailAddress $user.'EMail' -Title $user.'Job Title' -Country $user.region -OfficePhone $user.phone -AccountPassword(ConvertTo-SecureString "Qwer1234" -AsPlainText -Force) -Enabled $true
Add-ADGroupMember -Identity $user.'Group Member' -Members $user.Username
}
