特征值法求权重

[V,D] = eig(A)
% V: 特征向量
% D: 特征值
max_eig = max(D(:))
[r,c] = find(D == max_eig, 1)
% r: 行     c: 列
ans = V(:,c)
ans ./ sum(ans)
%    0.2121
%    0.2121
%    0.0679
%    0.4114
%    0.0965

计算一致性比例CR

% 1. 求最大特征值
n = size(A,1);
[V,D] = eig(A);
max_eig = max(D(:));

% 2. 计算一致性指标CI
CI = (max_eig-n) / (n-1);

% 3. 生成平均随机一致性指标RI
RI = [0 0.001 0.52 0.89 1.12 1.26 1.36 1.41 1.46 1.49 1.52 1.54 1.56 1.58 1.59];

% 4. 计算一致性比例CR
CR = CI / RI(n)
