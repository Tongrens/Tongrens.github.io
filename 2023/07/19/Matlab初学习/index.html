常用函数

% ---------------------------求和---------------------------
E = [1,2,3]
sum(E)          % 6
E = [1;2;3]
sum(E)          % 6

E = [1,2;3,4;5,6]
%   1   2
%   3   4
%   5   6
sum(E)          % 9   12
sum(E,1)        % 9   12
sum(E,2)
%    3
%    7
%   11
sum(E(:))       % 21

% ---------------------------取值---------------------------
% A =
%    1.0000   1.0000   4.0000   0.3333   3.0000
%    1.0000   1.0000   4.0000   0.3333   3.0000
%    0.2500   0.2500   1.0000   0.3333   0.5000
%    3.0000   3.0000   3.0000   1.0000   3.0000
%    0.3333   0.3333   2.0000   0.3333   1.0000
A(3,2)      % 0.25
A(2,:)      % 1.0000    1.0000    4.0000    0.3333    3.0000
A(:,3)
%      4
%      4
%      1
%      3
%      2

A([2,5],:)
%     1.0000    1.0000    4.0000    0.3333    3.0000
%     0.3333    0.3333    2.0000    0.3333    1.0000
A(2:3,:)
%     1.0000    1.0000    4.0000    0.3333    3.0000
%     0.2500    0.2500    1.0000    0.3333    0.5000
A(2:2:5,:)
%     1.0000    1.0000    4.0000    0.3333    3.0000
%     3.0000    3.0000    3.0000    1.0000    3.0000

% 矩阵(start:step:end,start:step:end)
% start默认为1, step默认为1, end默认为end

% -----------------------获取矩阵大小-----------------------
% A =
%      1     2     3
%      4     5     6
size(A)     % 2     3
[r,c] = size(A)
size(A,1)   % 2
size(A,2)   % 3

% ------------------------复制矩阵-------------------------
% A =
%      1     2     3
%      4     5     6
repmat(A,3,2)
%      1     2     3     1     2     3
%      4     5     6     4     5     6
%      1     2     3     1     2     3
%      4     5     6     4     5     6
%      1     2     3     1     2     3
%      4     5     6     4     5     6

% ---------------------------运算---------------------------
% A =
%      1     2
%      3     4
% B =
%      1     0
%      1     1

A * B
%      3     2
%      7     4
A / B
%    -1     2
%    -1     4
A .* B
%      1     0
%      3     4
A .* 2
%      2     4
%      6     8
A .^ 2
%      1     4
%      9    16

% ---------------------------查找---------------------------
% 返回向量或者矩阵中不为0的元素的位置索引
% X =
%      1     0     4    -3     0     0     0     8     6
find(X)     % 1     3     4     8     9
find(X,2)   % 1     3

% X =
%      1    -3     0
%      0     0     8
%      4     0     6
find(X)
%      1
%      3
%      4
%      8
%      9

% ---------------------------判断---------------------------
% X =
%      1    -3     0
%      0     0     8
%      4     0     6

X > 0
%    1   0   0
%    0   0   1
%    1   0   1
X == 4
%    0   0   0
%    0   0   0
%    1   0   0

求特征值和特征向量

% A =
%      1     2     3
%      2     2     1
%      2     0     3

E=eig(A)        % 求矩阵A的全部特征值，构成向量E
%    -1.3166
%     5.3166
%     2.0000

[V,D]=eig(A)    % 求矩阵A的全部特征值，构成对角阵D，并求A的特征向量构成V的列向量
% V =
%     0.8364   -0.6336   -0.2408
%    -0.3875   -0.5470   -0.8427
%    -0.3875   -0.5470    0.4815
% D =
%    -1.3166         0         0
%          0    5.3166         0
%          0         0    2.0000
