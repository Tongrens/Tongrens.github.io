排序算法-其他排序

2022/10/10 笔记 python sort

计数排序

def count_sort(lst, max_count=100):
    count = [0 for _ in range(max_count + 1)]
    for val in lst:
        count[val] += 1
    lst.clear()
    for ind, val in enumerate(count):
        for i in range(val):
            lst.append(ind)

li = [4, 6, 2, 3, 1, 5, 7, 9, 8]
count_sort(li, 9)
print(li)  # [1, 2, 3, 4, 5, 6, 7, 8, 9]

时间复杂度：O(n)
空间复杂度：O(n)
稳定性：稳定

基数排序

def radix_sort(lst):
    max_num = max(lst)
    it = 0
    while 10 ** it <= max_num:
        buckets = [[] for _ in range(10)]
        for i in lst:
            buckets[i // 10 ** it % 10].append(i)
        lst.clear()
        for bucket in buckets:
            lst.extend(bucket)
        it += 1

li = [4, 6, 2, 3, 1, 5, 7, 9, 8]
radix_sort(li)
print(li)

时间复杂度：O(nm)
空间复杂度：O(n)
稳定性：稳定

希尔排序

def insert_sort_gap(lst, gap):
    for i in range(gap, len(lst)):
        tmp = lst[i]
        j = i - gap
        while j >= 0 and lst[j] > tmp:
            lst[gap + j] = lst[j]
            j -= gap
        lst[j + gap] = tmp

def shell_sort(lst):
    n = len(lst) // 2
    while n >= 1:
        insert_sort_gap(lst, n)
        n //= 2

li = [4, 6, 2, 3, 1, 5, 7, 9, 8]
shell_sort(li)
print(li)  # [1, 2, 3, 4, 5, 6, 7, 8, 9]

时间复杂度：O(nlogn)
空间复杂度：O(1)
稳定性：不稳定

桶排序

def bucket_sort(lst, n=100, max_num=10000):
    buckets = [[] for _ in range(n)]
    for val in lst:
        i = min(val // (max_num // n), n - 1)
        buckets[i].append(val)
        for j in range(len(buckets[i]) - 1, 0, -1):
            if buckets[i][j - 1] > buckets[i][j]:
                buckets[i][j - 1], buckets[i][j] = buckets[i][j], buckets[i][j - 1]
            else:
                break
    sorted_lst = []
    for bucket in buckets:
        sorted_lst.extend(bucket)
    return sorted_lst

li = [4, 6, 2, 3, 1, 5, 7, 9, 8]
print(bucket_sort(li))  # [1, 2, 3, 4, 5, 6, 7, 8, 9]

时间复杂度：O(nlogm)
空间复杂度：O(logm)
稳定性：稳定
