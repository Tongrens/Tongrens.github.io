# BST
class BiTreeNode:
    def __init__(self, data):
        self.data = data
        self.lchild = None
        self.rchild = None
        self.parent = None

class BST:
    def __init__(self, lst=None):
        self.root = None
        if lst:
            for val in lst:
                self.insert(val)

    def insert(self, val):
        p = self.root
        if not p:
            self.root = BiTreeNode(val)
            return
        while True:
            if val < p.data:
                if p.lchild:
                    p = p.lchild
                else:
                    p.lchild = BiTreeNode(val)
                    p.lchild.parent = p
                    return
            elif val > p.data:
                if p.rchild:
                    p = p.rchild
                else:
                    p.rchild = BiTreeNode(val)
                    p.rchild.parent = p
                    return
            else:
                return

    def find(self, val):
        p = self.root
        while p:
            if val > p.data:
                p = p.rchild
            elif val < p.data:
                p = p.lchild
            else:
                return p
        return None

    def __delete_node_1(self, node):  # 删除叶子节点
        if not node.parent:     # 只有一个根节点
            self.root = None
        elif node == node.parent.lchild:
            node.parent.lchild = None
        else:
            node.parent.rchild = None

    def __delete_node_2_left(self, node):  # 删除非叶子节点，只有左子树
        if not node.parent:
            self.root = node.lchild
            node.lchild.parent = None
        elif node == node.parent.lchild:
            node.parent.lchild = node.lchild
            node.lchild.parent = node.parent
        else:
            node.parent.rchild = node.lchild
            node.lchild.parent = node.parent

    def __delete_node_2_right(self, node):  # 删除非叶子节点，只有右子树
        if not node.parent:
            self.root = node.rchild
            node.rchild.parent = None
        elif node == node.parent.lchild:
            node.parent.lchild = node.rchild
            node.rchild.parent = node.parent
        else:
            node.parent.rchild = node.rchild
            node.rchild.parent = node.parent

    def delete(self, val):
        if self.root:
            node = self.find(val)
            if not node:
                raise ValueError('\"' + str(val) + '\" not in tree')
            if not node.lchild and not node.rchild:
                self.__delete_node_1(node)
            elif not node.rchild:
                self.__delete_node_2_left(node)
            elif not node.lchild:
                self.__delete_node_2_right(node)
            else:
                min_node = node.rchild
                while min_node.lchild:
                    min_node = min_node.lchild
                node.data = min_node.data
                if min_node.rchild:
                    self.__delete_node_2_right(min_node)
                else:
                    self.__delete_node_1(min_node)

    def order(self, root):
        if root:
            self.order(root.lchild)
            print(root.data, end=',')
            self.order(root.rchild)
