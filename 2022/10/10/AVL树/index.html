# AVLTree
from BST import BiTreeNode, BST

class AVLNode(BiTreeNode):
    def __init__(self, data):
        BiTreeNode.__init__(self, data)
        self.bf = 0

class AVLTree(BST):
    def __init__(self, li=None):
        BST.__init__(self, li)

    def rotate_left(self, p, c):
        lc = c.lchild
        p.rchild = lc
        if lc:
            lc.parent = p
        c.lchild = p
        p.parent = c
        p.bf = 0
        c.bf = 0
        return c

    def rotate_right(self, p, c):
        rc = c.rchild
        p.lchild = rc
        if rc:
            rc.parent = p
        c.rchild = p
        p.parent = c
        p.bf = 0
        c.bf = 0
        return c

    def rotate_right_left(self, p, c):
        g = c.lchild
        rrc = g.rchild
        c.lchild = rrc
        if rrc:
            rrc.parent = c
        g.rchild = c
        c.parent = g
        rlc = g.lchild
        p.rchild = rlc
        if rlc:
            rlc.parent = p
        g.lchild = p
        p.parent = g
        if g.bf > 0:
            p.bf = -1
            c.bf = 0
        elif g.bf < 0:
            p.bf = 0
            c.bf = 1
        else:
            p.bf = 0
            c.bf = 0
        return g

    def rotate_left_right(self, p, c):
        g = c.rchild
        rlc = g.lchild
        c.rchild = rlc
        if rlc:
            rlc.parent = c
        g.lchild = c
        c.parent = g
        rrc = g.rchild
        p.lchild = rrc
        if rrc:
            rrc.parent = p
        g.rchild = p
        p.parent = g
        if g.bf > 0:
            p.bf = 0
            c.bf = -1
        elif g.bf < 0:
            p.bf = 1
            c.bf = 0
        else:
            p.bf = 0
            c.bf = 0
        return g

    def insert(self, val):
        p = self.root
        if not p:
            self.root = AVLNode(val)
            return
        while True:
            if val < p.data:
                if p.lchild:
                    p = p.lchild
                else:
                    p.lchild = AVLNode(val)
                    p.lchild.parent = p
                    node = p.lchild
                    break
            elif val > p.data:
                if p.rchild:
                    p = p.rchild
                else:
                    p.rchild = AVLNode(val)
                    p.rchild.parent = p
                    node = p.rchild
                    break
            else:
                return
        while node.parent:
            g = node.parent.parent
            x = node.parent
            if node == node.parent.lchild:
                if node.parent.bf < 0:
                    if node.bf > 0:
                        n = self.rotate_left_right(node.parent, node)
                    else:
                        n = self.rotate_right(node.parent, node)
                elif node.parent.bf == 0:
                    node.parent.bf = -1
                    node = node.parent
                    continue
                else:
                    node.parent.bf = 0
                    break
            else:
                if node.parent.bf > 0:
                    if node.bf < 0:
                        n = self.rotate_right_left(node.parent, node)
                    else:
                        n = self.rotate_left(node.parent, node)
                elif node.parent.bf == 0:
                    node.parent.bf = 1
                    node = node.parent
                    continue
                else:
                    node.parent.bf = 0
                    break
            n.parent = g
            if g:
                if x == g.lchild:
                    g.lchild = n
                else:
                    g.rchild = n
                break
            else:
                self.root = n
                break

tree = AVLTree([9,8,7,6,5,4,3,2,1])
tree.order(tree.root)
print()
tree.insert(10)
tree.order(tree.root)
