# 链表
class Node:
    def __init__(self, item=None):
        self.item = item
        self.next = None
        self.prior = Node  # 双链表

# 头插法
def create_linklist_head(lst):
    head = Node(lst[0])
    for element in lst[1:]:
        node = Node(element)
        head.prior = node  # 双链表
        node.next = head
        head = node
    return head

# 尾插法
def create_linklist_tail(lst):
    head = Node(lst[0])
    tail = head
    for element in lst[1:]:
        node = Node(element)
        tail.next = node
        tmp = tail
        tail = node
        tail.prior = tmp  # 26&28双链表
    return head

# 输出链表
def print_linklist(lk):
    while lk:
        print(lk.item, end=',')
        lk = lk.next
    print()

# 头插法eg
lk1 = create_linklist_head([1, 2, 3])
print(lk1.item)  # 3
print(lk1.next.item)  # 2
print(lk1.next.prior.item)  # 3(双链表)
print_linklist(lk1)  # 3,2,1,
# 尾插法eg
lk2 = create_linklist_tail([1, 2, 3])
print(lk2.item)  # 1
print(lk2.next.item)  # 2
print(lk2.next.prior.item)  # 1(双链表)
print_linklist(lk2)  # 1,2,3,
# 插入元素
p = Node(5)
p.next = lk2.next
lk2.next.prior = p
p.prior = lk2   # 54&55双链表
lk2.next = p
print_linklist(lk2)  # 1,5,2,3,
# 删除元素
lk2.next.next.prior = lk2
lk2.next = lk2.next.next
print_linklist(lk2)  # 1,2,3,
"""
————————————时间复杂度——————————————
             列表      链表
按元素查找：   O(n)      O(n)
按下标查找：   O(1)      O(n)
插入&删除：   O(n)      O(1)
"""
