# HashTable
class LinkList:
    class Node:
        def __init__(self, item=None):
            self.item = item
            self.next = None

    class LinkListIterator:
        def __init__(self, head):
            self.current = head

        def __iter__(self):
            return self

        def __next__(self):
            if self.current is None:
                raise StopIteration
            else:
                item = self.current.item
                self.current = self.current.next
                return item

    def __init__(self, iterable=None):
        self.head = None
        self.tail = None
        if iterable:
            self.extend(iterable)

    def append(self, obj):
        node = self.Node(obj)
        if self.head is None:
            self.head = node
            self.tail = node
        else:
            self.tail.next = node
            self.tail = node

    def extend(self, iterable):
        for item in iterable:
            self.append(item)

    def find(self, obj):
        for item in self:
            if item == obj:
                return True
        return False

    def __iter__(self):
        return self.LinkListIterator(self.head)

    def __repr__(self):
        return str(list(self))

class HashTable:
    def __init__(self, size=101):
        self.size = size
        self.table = [LinkList() for i in range(self.size)]

    def hash(self, obj):
        return obj % self.size

    def insert(self, obj):
        index = self.hash(obj)
        if self.find(obj):
            return False
        else:
            self.table[index].append(obj)
            return True

    def find(self, obj):
        index = self.hash(obj)
        return self.table[index].find(obj)

ht = HashTable()
ht.insert(0)
ht.insert(1)
ht.insert(101)
print(','.join(map(str, ht.table)))
print(ht.find(1))
print(ht.find(2))
