def fun1(n, m):
    if n == m == 1: return 1
    ans, flag, lst = 1, [1, 1], [[1]]
    while True:
        for i in range(len(lst)):
            if i == 0: flag[1], ans = flag[1] + 1, ans + 1
            else: flag[0], flag[1], ans = flag[0] + 1, flag[1] - 1, ans + 1
            if flag[0] == n and flag[1] == m: return ans
            lst[i].append(ans)
        flag[0], flag[1], ans = flag[0] + 1, flag[1] - 1, ans + 1
        if flag[0] == n and flag[1] == m: return ans
        lst.append([ans])
        flag[0], ans = flag[0] + 1, ans + 1
        if flag[0] == n and flag[1] == m: return ans
        lst.append([ans])
        for i in range(len(lst) - 2, -1, -1):
            flag[0], flag[1], ans = flag[0] - 1, flag[1] + 1, ans + 1
            if flag[0] == n and flag[1] == m: return ans
            lst[i].append(ans)
n, m = map(int, input().split())
print(fun1(n, m))  # input(2022 2022) output(8156761)

2、3、5、7、11、13是素数序列。 类似：7、37、67、97、127、157这样完全由素数组成的等差数列，叫等差素数数列。
上边的数列公差为30，长度为 6。
2004年，格林与华人陶哲轩合作证明了：存在任意长度的素数等差数列。 这是数论领域一项惊人的成果！
有这一理论为基础，请你借助手中的计算机，满怀信心地搜索：
长度为10的等差素数列，其公差最小值是多少？

def fun2(a):
    def isPrimesl(n):
        if n == 1:
            return False
        for i in range(2, int(n ** 0.5) + 1):
            if n % i == 0:
                return False
        return True
    num = []
    for ans in range(3, 1000, 2):
        for i in range(1, 1000):    # 等差至多999
            tmp = int(str(ans))
            for _ in range(a-1):
                if not isPrimesl(tmp + i):
                    break
                tmp += i
            else:
                num.append(i)
    return min(num)
a = int(input())
print(fun2(a))  # input(10) output(210) first_num(199)

def fun4(n):
    if n == 1:
        return 1
    lst = [[1]]
    ans = [0, 1]
    while True:
        tmp = [0]
        tmp.extend(lst[-1])
        tmp.append(0)
        for i in range(len(tmp) - 1):
            tmp[i] = tmp[i] + tmp[i + 1]
        lst.append(tmp[:-1])
        ans[0] += 1
        ans[1] += ans[0]
        if n in lst[-1]:
            ans[1] += lst[-1].index(n)
            return ans[1]
n = int(input())
print(fun4(n))  # input(6) output(13)

def fun5(a):
    def isPrimesl(n):
        if n == 1:
            return False
        for i in range(2, int(n ** 0.5) + 1):
            if n % i == 0:
                return False
        return True
    flag = 0

小蓝发现，他将 1至 1000000007之间的不同的数与 2021相乘后再求除以 1000000007的余数，会得到不同的数。 小蓝想知道，能不能在 1 至 1000000007之间找到一个数，与 2021相乘后 再除以 1000000007后的余数为 999999999。如果存在，请在答案中提交这个数； 如果不存在，请在答案中提交 0。

def fun7():
    for i in range(1, 1000000008):
        if (i * 2021) % 1000000007 == 999999999:
            return i
    else:
        return 0

记 f(x)为 x的所有因数的平方的和。例如: f(12)=$1^2+2^2+3^2+4^2+6^2+12^2$
定义$g(n)=∑^n_{i=1}f(i)$。给定 n, 求g(n)除以$10^9+7$的余数。

输入格式
输入一行包含一个正整数 n。
输出格式
输出一个整数表示答案g(n)除以$10^9+7$的余数。
例如：
输入100000
输出680584257

def fun8(n):
    def f(x):
        nums = []
        for i in range(1, int(x ** 0.5) + 1):
            if x % i == 0:
                if i not in nums:
                    nums.append(i)
                if x // i not in nums:
                    nums.append(x // i)
        return sum([i ** 2 for i in nums])
    ans = 0
    for i in range(1, n + 1):
        tmp = f(i)
        ans += tmp
    return ans % (10 ** 9 + 7)
n = int(input())
print(fun8(n))  # input(100000) output(680584257)
